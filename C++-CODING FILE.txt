#include <iostream>
#include <stack>
#include <string>

using namespace std;

// Helper to show the current state of our stacks
void displayStatus(const string& current, stack<string> back, stack<string> forward) {
    cout << "\n--- Currently on: " << current << " ---" << endl;
    cout << "History (Back): " << (back.empty() ? "[Empty]" : "[Available]") << endl;
    cout << "Next (Forward): " << (forward.empty() ? "[Empty]" : "[Available]") << endl;
}

int main() {
    // Step 2: Create Back stack and Forward stack
    stack<string> back_stack;
    stack<string> forward_stack;
    
    // Step 3: Open Home Page
    string current_page = "Home Page";
    int choice;

    while (true) {
        displayStatus(current_page, back_stack, forward_stack);
        cout << "\n1. Open New Page\n2. Back\n3. Forward\n4. Exit\nSelect an action: ";
        cin >> choice;

        if (choice == 1) {
            // Step 4: Store current page in Back stack
            back_stack.push(current_page);
            // Step 5: Clear Forward stack
            while (!forward_stack.empty()) forward_stack.pop();
            // Step 6: Open new page
            cout << "Enter page name: ";
            cin.ignore(); // Clear buffer
            getline(cin, current_page);
        }
        else if (choice == 2) {
            // Step 7: Move current page to Forward stack
            if (!back_stack.empty()) {
                forward_stack.push(current_page);
                // Step 8: Open previous page from Back stack
                current_page = back_stack.top();
                back_stack.pop();
            } else {
                cout << "!! No history to go back to." << endl;
            }
        }
        else if (choice == 3) {
            // Step 9: Move current page to Back stack
            if (!forward_stack.empty()) {
                back_stack.push(current_page);
                // Step 10: Open next page from Forward stack
                current_page = forward_stack.top();
                forward_stack.pop();
            } else {
                cout << "!! No forward history." << endl;
            }
        }
        else if (choice == 4) {
            // Step 12: Stop
            cout << "Exiting browser simulation..." << endl;
            break;
        }
        else {
            cout << "Invalid choice. Please enter 1-4." << endl;
        }
    }
    return 0;
}
